Q2:Find the Fives

Answer:

{
let count=0;
while(N!==0){
let lastdigit=N%10;
if(lastdigit===5){
count++;
}
N=Math.floor(N/10);
}
return count;
};

Q4:Find the digits sum of a number

Answer:

const Number_Sum = (N) =>
{
let sum=0;
while(N!==0){
let digit=N%10;
sum=sum+digit;
N=Math.floor(N/10);
}
return sum;
};

Q7:Check if the number is Prime or not

Answer:

{

if(N%i===0){
count++;
}
}
